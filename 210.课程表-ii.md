/*
第一种方法:利用dfs搜索算法,倒序得到正确的拓扑排序其中的一种.

主函数中,对所有没有搜索过的点进行遍历dfs(以防图是裂开的)

dfs():
先将该点置为搜索中,即1,
然后对其周围邻接点(有向图 表示 预修->准修 的关系)进行遍历
如果没有搜索过,即为0,则对该点进行dfs,
    如果为1,表示邻接点正在搜索中,又回到了该点,因此有环存在,改图非法
    如果为2,则直接跳过该点(函数中没写出来)
    如果为0,表示该邻接点没有搜索过,对该点进行dfs();
*/

// @lc code=start
class Solution {
public:
    vector<int> findOrder(int numCourses, vector<vector<int>>& prerequisites) {
        build_graph(numCourses, prerequisites);
        for (int i = 0; i < numCourses && !invalid; ++i){
            if (!visited[i]){
                dfs(i);
            }
        }
        if (invalid) {
            return {};
        }
        reverse(res.begin(), res.end());
        return res;
    }
private:
    vector<int> visited;
    vector<vector<int>> graph;
    bool invalid{false};
    vector<int> res;
    void dfs(int course){
        visited[course] = 1;

        for (int v : graph[course]){

            if (visited[v] == 0){
                dfs(v);
                if (invalid) {
                    return;
                }
            }

            if (visited[v] == 1){
                invalid = true;
                return;
            }
        }
        visited[course] = 2;
        res.push_back(course);
    }
    void build_graph(int num, vector<vector<int>>& prerequisites){
        graph.resize(num);
        visited.resize(num);
        for (auto pre : prerequisites){
            graph[pre[1]].push_back(pre[0]);
        }
    }
};

/*
bfs()广度优先搜索
*/

class Solution {
public:
    vector<int> findOrder(int numCourses, vector<vector<int>>& prerequisites) {
        build_graph(numCourses, prerequisites);
        bfs(numCourses);
        return res.size() == numCourses ? res : (vector<int> {});
    }
private:
    vector<bool> visited;
    vector<vector<int>> graph;
    vector<int> res;
    queue<int> que;
    vector<int> indegree;
    void bfs(int n){

        for (int i = 0; i < n; ++i){
            if (indegree[i] == 0) {
                que.push(i);
            }
        }
        while (!que.empty()){
            auto u = que.front();
            res.push_back(u);
            que.pop();
            for (auto v : graph[u]){
                if (--indegree[v] == 0) {
                    que.push(v);
                }
            }
        }
    }

    void build_graph(int num, vector<vector<int>>& prerequisites){
        graph.resize(num);
        visited.resize(num);
        indegree.resize(num);
        for (auto pre : prerequisites){
            graph[pre[1]].push_back(pre[0]);
            ++indegree[pre[0]];
        }
    }
};
