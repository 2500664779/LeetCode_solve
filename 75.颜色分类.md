#include"all_include"
// @lc code=start
class Solution {
public:
    void sortColors(vector<int>& nums) {
        if (nums.size() <= 1){
            return;
        }
        int left = -1, right = nums.size();
        int i = 0;
        while (i < right){
            if (nums[i] == 1){
                ++i;
            }
            if (nums[i] == 0){
                swap(nums[i++], nums[++left]);
                continue;
            }
            if (nums[i] == 2){
                swap(nums[i], nums[--right]);
            } 
        }
    }
};
