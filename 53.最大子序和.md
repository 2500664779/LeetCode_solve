 * [53] 最大子序和
 */

#include"all_include"
// @lc code=start
class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int *p = new int[nums.size()];
        int res = INT_MIN;
        for (int i = 0; i < nums.size(); ++i){
            if (i == 0 || nums[i - 1] < 0)
                p[i] = nums[i];
            else {
                p[i] = nums[i] + p[i - 1];
            }
            res = max(res, p[i]);
        }
        return res;
    }
};
// @lc code=end
